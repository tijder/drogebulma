{{- /* Markdown image render hook: makes body images zoomable via the no-JS lightbox. Renders the standard lazy (resolving bundle resources for intrinsic width/height to avoid layout shift) and wraps it in `lightbox.html` so a click enlarges it to a full-viewport overlay. The wrapper id is the image ordinal, which is unique within the page. */ -}} {{- $alt := .PlainText | default .Title -}} {{- $dest := .Destination -}} {{- $img := "" -}} {{- with .Page.Resources.GetMatch $dest -}} {{- $img = printf `` .RelPermalink $alt (string .Width) (string .Height) -}} {{- else -}} {{- $img = printf `` $dest $alt -}} {{- end -}} {{- partial "lightbox.html" (dict "id" (printf "lb-%d" .Ordinal) "inner" $img "label" $alt) -}}
