{{- /* Renders a statistics dashboard aggregated from the generated trophies game pages (their params), so it never touches the raw 6 MB data again. @example: {{ partial "trophy/stats.html" . }} */ -}} {{- $games := where site.RegularPages "Section" "trophies" }} {{- $earned := 0 }}{{ $total := 0 }} {{- /* Source-level aggregates: count + sum of per-source completion percentages, so "100% completed" and "average progress" are measured per platform source. */}} {{- $sourceCount := 0 }}{{ $sourcesDone := 0 }}{{ $pctSum := 0 }} {{- range $games }} {{- $earned = add $earned (.Params.earned | default 0) }} {{- $total = add $total (.Params.total | default 0) }} {{- range .Params.sources }} {{- $sourceCount = add $sourceCount 1 }} {{- $pctSum = add $pctSum (.pct | default 0) }} {{- if ge (.pct | default 0) 100 }}{{ $sourcesDone = add $sourcesDone 1 }}{{ end }} {{- end }} {{- end }} {{- $avgPct := 0 }}{{ if gt $sourceCount 0 }}{{ $avgPct = int (math.Round (math.Div $pctSum $sourceCount)) }}{{ end }}

{{ i18n "trophyGames" }}

{{ len $games }}

{{ i18n "totalEarned" }}

{{ $earned }}/{{ $total }}

{{ i18n "sourcesCompleted" }}

{{ $sourcesDone }}/{{ $sourceCount }}

{{ i18n "averageProgress" }}

{{ $avgPct }}%

{{ i18n "byPlatform" }}

{{- range $pf := slice "steam" "psn" }} {{- $pe := 0 }}{{ $pt := 0 }} {{- range $games }}{{ $pe = add $pe (index .Params.platformEarned $pf | default 0) }}{{ $pt = add $pt (index .Params.platformTotal $pf | default 0) }}{{ end }} {{- if gt $pt 0 }} {{ i18n (printf "platform_%s" $pf) | default $pf }}: {{ $pe }}/{{ $pt }} {{- end }} {{- end }}

{{ i18n "byType" }}

{{- range $ty := slice "platinum" "gold" "silver" "bronze" "achievement" }} {{- $te := 0 }}{{ $tt := 0 }} {{- range $games }}{{ $te = add $te (index .Params.typeEarned $ty | default 0) }}{{ $tt = add $tt (index .Params.typeTotal $ty | default 0) }}{{ end }} {{- if gt $tt 0 }} {{ i18n (printf "trophyType_%s" $ty) | default $ty }}: {{ $te }}/{{ $tt }} {{- end }} {{- end }}
